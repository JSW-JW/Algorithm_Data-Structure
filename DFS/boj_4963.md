```python
import sys
input = sys.stdin.readline
sys.setrecursionlimit(10**7)

ways = {'E': (0, 1),
        'W': (0, -1),
        'S': (1, 0),
        'N': (-1, 0),
        'EU': (-1, 1),
        'ED': (1, 1),
        'WU': (-1, -1),
        'WD': (1, -1)}

def dfs(info, pos):
    x,y = pos
    if info[x][y] == 0 or visited[x][y] == True: return
    visited[x][y] == True
    
    for dir_ in ways.values():
        a, b = dir_
        if 0 <= y+b < len(info[0]) and 0 <= x+a < len(info):
            dfs(info, (a, b))
while True:
    a, b = map(int, input().split())
    if (a,b) == (0, 0): break
    info = [list(map(int, input().split())) for _ in range(b)]
    visited = [[False] * a for _ in range(b)]
    answer = 0
    
    for i in range(b):
        for j in range(a):
            if visited[i][j] == False:
                dfs(info, (i,j))
                answer += 1
            
    print(answer)
    
```

```python
import sys
input = sys.stdin.readline
sys.setrecursionlimit(10**7)

ways = {'E': (0, 1),
        'W': (0, -1),
        'S': (1, 0),
        'N': (-1, 0),
        'EU': (-1, 1),
        'ED': (1, 1),
        'WU': (-1, -1),
        'WD': (1, -1)}

def dfs(info, x, y):
    if info[x][y] == 0 or visited[x][y] == True: return
    visited[x][y] == True
    
    for dir_ in ways.values():
        if 0 <= y + dir_[1] < len(info[0]) and 0 <= x + dir_[0] < len(info):
            dfs(info, dir_[0], dir_[1])
            
while True:
    a, b = map(int, input().split())
    if (a,b) == (0, 0): break
    info = [list(map(int, input().split())) for _ in range(b)]
    visited = [[False] * a for _ in range(b)]
    answer = 0
    
    for i in range(b):
        for j in range(a):
            if visited[i][j] == False:
                dfs(info, i, j)
                answer += 1
            
    print(answer)
```
