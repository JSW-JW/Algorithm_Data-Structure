# 백준 1935 후위 표기식2

```python
import sys
input = sys.stdin.readline
from collections import defaultdict
import string

n = int(input())
s = [i for i in input().strip()]
d = defaultdict(int)

st = []
ascii = list(string.ascii_uppercase)
for i in range(n):
    num = int(input())
    d[ascii[i]] = num

for i in range(len(s)):
    if 'A' <= s[i] <= 'Z': st.append(d[s[i]])
    else:
        if s[i] == '*':
            a,b = st.pop(), st.pop()
            st.append(b * a)
        elif s[i] == '+':
            a,b = st.pop(), st.pop()
            st.append(b + a)
        elif s[i] == '/':
            a,b = st.pop(), st.pop()
            st.append(b / a)
        elif s[i] == '-':
            a,b = st.pop(), st.pop()
            st.append(b - a)

print("{:.2f}".format(st[0]))
```
