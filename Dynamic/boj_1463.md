## RTE(TypeError)
```python
import sys
input = sys.stdin.readline

N = int(input())
d = [0] * (N+1)
d[1], d[2], d[3] = 0, 1, 1

for i in range(4, N+1):
    if i % 2 != 0 and i % 3 != 0:
        d[i] = d[i-1] + 1
    elif i % 2 == 0 and i % 3 != 0:
        d[i] = min(d[i/2], d[i-1])
    elif i % 2 != 0 and i % 3 == 0:
        d[i] = min(d[i/3], d[i-1])
    elif i % 2 == 0 and i % 3 == 0:
        d[i] = min(d[i/3], d[i/2], d[i-1])
        
print(d[N])
```


## RTE(IndexError)
```
import sys
input = sys.stdin.readline

N = int(input())
d = [0] * (N+1)
d[1], d[2], d[3] = 0, 1, 1

for i in range(4, N+1):
    if i % 2 != 0 and i % 3 != 0:
        d[i] = d[i-1] + 1
    elif i % 2 == 0 and i % 3 != 0:
        d[i] = min(d[i//2], d[i-1]) + 1
    elif i % 2 != 0 and i % 3 == 0:
        d[i] = min(d[i//3], d[i-1]) + 1
    elif i % 2 == 0 and i % 3 == 0:
        d[i] = min(d[i//3], d[i//2], d[i-1]) + 1
        
print(d[N])
```
