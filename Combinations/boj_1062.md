```python
import sys
import string
input = sys.stdin.readline
from collections import Counter

# anta 
# tica  -> a, n, t, i, c
# 1 <= N <= 50,  0 <= K < 26

answer = 0
learned = {'a', 'n', 't', 'i', 'c'}
nope = {'a', 'n', 't', 'i', 'c'}

l = dict.fromkeys(string.ascii_lowercase, 0)
for key in nope:
    del l[key]
    
N, K = map(int, input().split())
words = []
for _ in range(N):
    word = input().strip()
    tmp = word[4:-4]
    words.append(tmp)


for ch in l:
    idx = 0
    for word in words:
        if ch in word: idx += 1
    l[ch] = idx      
    
result = sorted(l.items(), key=lambda item: -item[1])

for i in range(K-5) :
    ch = result[i][0]
    learned.add(ch)
    
    
for word in words:
    capable = True
    for ch in word:
        if ch not in learned:
            capable = False
    
    if capable is True:
        answer += 1
        
if K < 5:
    print(0)
else:
    print(answer)
```


### 새로운 코드로 시도 (combinations + back tracking) -> 시간초과

```python
import sys
import string
input = sys.stdin.readline
from collections import Counter
from itertools import combinations

# anta 
# tica  -> a, n, t, i, c
# 1 <= N <= 50,  0 <= K < 26


def countReadableWords(words, learned):
    cnt = 0
    for word in words:
        capable = True
        for ch in word:
            if learned[ch] == 0:
                capable = False
        if capable is True: cnt += 1
    return cnt


def solution():
    answer = 0
    fixed = {'a', 'n', 't', 'i', 'c'}
    learned = dict.fromkeys(string.ascii_lowercase, 0)
    alphabets = dict.fromkeys(string.ascii_lowercase, 0)

    N, K = map(int, input().split())
    words = []
    for _ in range(N):
        word = input().strip()
        tmp = word[4:-4]
        words.append(tmp)
    
    for t in fixed:
        learned[t] = 1
        
    for teach in combinations(alphabets, K-5):
        for t in teach:
            learned[t] = 1
            cnt = countReadableWords(words, learned)
            answer = max(answer, cnt)
            
        for t in teach:
            if t not in fixed:
                learned[t] = 0
            
    if K < 5:
        print(0)
    elif len(alphabets) < K-5:
        print(N)
    else:
        print(answer)
        
        
solution()
```
