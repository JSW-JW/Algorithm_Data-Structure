# 백준 1062번 가르침

### 오답 코드 ( 반례: https://ideone.com/yRka2J)

```python
import sys
import string
input = sys.stdin.readline
from collections import Counter

# anta 
# tica  -> a, n, t, i, c
# 1 <= N <= 50,  0 <= K < 26

answer = 0
learned = {'a', 'n', 't', 'i', 'c'}
nope = {'a', 'n', 't', 'i', 'c'}

l = dict.fromkeys(string.ascii_lowercase, 0)
for key in nope:
    del l[key]
    
N, K = map(int, input().split())
words = []
for _ in range(N):
    word = input().strip()
    tmp = word[4:-4]
    words.append(tmp)


for ch in l:
    idx = 0
    for word in words:
        if ch in word: idx += 1
    l[ch] = idx      
    
result = sorted(l.items(), key=lambda item: -item[1])

for i in range(K-5) :
    ch = result[i][0]
    learned.add(ch)
    
    
for word in words:
    capable = True
    for ch in word:
        if ch not in learned:
            capable = False
    
    if capable is True:
        answer += 1
        
if K < 5:
    print(0)
else:
    print(answer)
```
