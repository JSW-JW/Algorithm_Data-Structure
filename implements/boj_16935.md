class ArrConverter:
    def __init__(self, arr):
        self.arr = arr

    def func1(self, arr):
        return arr[::-1]

    def func2(self, arr):
        n = len(arr[0])
        half = len(arr[0]) // 2
        for i in range(len(arr)):
            for j in range(half):
                arr[i][j], arr[i][n-1-j] = arr[i][n-1-j], arr[i][j]
        return arr

    def func3(self, arr):
        result = []
        for i in range(len(arr[0])):
            tmp = []
            for j in range(len(arr)-1, -1, -1):
                tmp.append(arr[j][i])
            result.append(tmp)
        return result

    def func4(self, arr):
        result = []
        for i in range(len(arr[0])-1, -1, -1):
            tmp = []
            for j in range(len(arr)):
                tmp.append(arr[j][i])
            result.append(tmp)
        return result

    def func5(self, arr):
        n = len(arr)
        m = len(arr[0])

        copied = [arr[i][::] for i in range(n)]
        # 1 -> 2
        for i in range(n // 2):
            for j in range(m // 2):
                arr[i][j + m//2] = copied[i][j]

        # 2 -> 3
        for i in range(n // 2):
            for j in range(m // 2, m):
                arr[i + n//2][j] = copied[i][j]

        # 3 -> 4
        for i in range(n // 2, n):
            for j in range(m // 2):
                arr[i][j] = copied[i][j + m // 2]

        # 4 -> 1
        for i in range(n // 2):
            for j in range(m // 2):
                arr[i][j] = copied[i + n//2][j]

        return arr



    def func6(self, arr):
        n = len(arr)
        m = len(arr[0])

        copied = [arr[i][::] for i in range(n)]
        # 1 -> 4
        for i in range(n // 2):
            for j in range(m // 2):
                arr[i + n//2][j] = copied[i][j]

        # 4 -> 3
        for i in range(n // 2, n):
            for j in range(m // 2):
                arr[i][j + m//2] = copied[i][j]

        # 3 -> 2
        for i in range(n // 2):
            for j in range(m // 2, m):
                arr[i][j] = copied[i + n//2][j]

        # 2 -> 1
        for i in range(n // 2):
            for j in range(m // 2):
                arr[i][j] = copied[i][j + m//2]

        return arr


if __name__ == "__main__":
    n, m, r = map(int, input().split())
    arr = [list(map(int, input().split())) for _ in range(n)]
    # print(*arr, sep='\n')
    cmds = list(map(int, input().split()))
    arrConverter = ArrConverter(arr)
    cmdMapper = {1: arrConverter.func1, 2: arrConverter.func2, 3: arrConverter.func3, 4: arrConverter.func4, 5: arrConverter.func5, 6: arrConverter.func6}
    for cmd in cmds:
        arr = cmdMapper[cmd](arr)
    for iter in arr:
        print(*iter, sep=' ')
