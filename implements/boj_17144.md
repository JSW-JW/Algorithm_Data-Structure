# 백준 17144 미세먼지 안녕!
```python
import sys
input = sys.stdin.readline

def sol(R, C, T, g):
    ac = []
    for i in range(R):
        for j in range(C):
            if g[i][j] == -1:
                ac.append((i, j))
    tmp = [[0] * C for _ in range(R)]
    def spread(i, j):
        dx, dy = [0, 1, 0, -1], [1, 0, -1, 0]
        cnt = 0
        for k in range(4):
            nx, ny = i + dx[k], j + dy[k]
            if 0 <= nx < R and 0 <= ny < C and g[nx][ny] != -1:
                cnt += 1
                tmp[nx][ny] += g[i][j] // 5
        g[i][j] -= cnt * (g[i][j] // 5)

    def operate_up():
        upper = ac[0]
        up_lst = []
        for i in range(1, C):
            up_lst.append((upper[0], i))
        for i in range(upper[0]-1, -1, -1):
            up_lst.append((i, C-1))
        for i in range(C-2, -1, -1):
            up_lst.append((0, i))
        for i in range(1, upper[0]):
            up_lst.append((i, 0))

        for i in range(len(up_lst)-1, 0, -1):
            cur = up_lst[i]
            prev = up_lst[i-1]
            g[cur[0]][cur[1]] = g[prev[0]][prev[1]]
        tmp = up_lst[0]
        g[tmp[0]][tmp[1]] = 0

    def operate_down():
        down = ac[1]
        down_lst = []
        for i in range(1, C):
            down_lst.append((down[0], i))
        for i in range(down[0]+1, R):
            down_lst.append((i, C-1))
        for i in range(C-2, -1, -1):
            down_lst.append((R-1, i))
        for i in range(R-2, down[0], -1):
            down_lst.append((i, 0))

        for i in range(len(down_lst)-1, 0, -1):
            cur = down_lst[i]
            prev = down_lst[i-1]
            g[cur[0]][cur[1]] = g[prev[0]][prev[1]]
        tmp = down_lst[0]
        g[tmp[0]][tmp[1]] = 0

    while T:
        for i in range(R):
            for j in range(C):
                if g[i][j] > 0:
                    spread(i, j)

        # 확산된 미세 먼지 총계에 대해 g 에 한번에 계산
        for i in range(R):
            for j in range(C):
                if tmp[i][j] != 0:
                    g[i][j] += tmp[i][j]
                    tmp[i][j] = 0
        operate_up()
        operate_down()
        T -= 1

    ans = 0
    for i in range(R):
        for j in range(C):
            if g[i][j] > 0:
                ans += g[i][j]
    print(ans)


R,C,T = map(int, input().split())
g = [list(map(int, input().split())) for _ in range(R)]
sol(R, C, T, g)
```
